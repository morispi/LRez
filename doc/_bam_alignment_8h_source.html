197 template <typename T>
198 bool BamAlignment::AddTag(const std::string& tag, const std::string& type, const T& value)
199 {
200 
201  // if char data not populated, do that first
202  if (SupportData.HasCoreOnly) BuildCharData();
203 
204  // check tag/type size
205  if (!IsValidSize(tag, type)) {
206  // TODO: set error string?
207  return false;
208  }
209 
210  // check that storage type code is OK for T
211  if (!TagTypeHelper<T>::CanConvertTo(type.at(0))) {
212  // TODO: set error string?
213  return false;
214  }
215 
216  // localize the tag data
217  char* pTagData = (char*)TagData.data();
218  const unsigned int tagDataLength = TagData.size();
219  unsigned int numBytesParsed = 0;
220 
221  // if tag already exists, return false
222  // use EditTag explicitly instead
223  if (FindTag(tag, pTagData, tagDataLength, numBytesParsed)) {
224  // TODO: set error string?
225  return false;
226  }
227 
228  // otherwise, convert value to string
229  union
230  {
231  T value;
232  char valueBuffer[sizeof(T)];
233  } un;
234  un.value = value;
235 
236  // copy original tag data to temp buffer
237  const std::string newTag = tag + type;
238  const std::size_t newTagDataLength =
239  tagDataLength + newTag.size() + sizeof(T); // leave room for new T
240  RaiiBuffer originalTagData(newTagDataLength);
241  std::memcpy(originalTagData.Buffer, TagData.c_str(),
242  tagDataLength + 1); // '+1' for TagData null-term
243 
244  // append newTag
245  std::strcat(originalTagData.Buffer + tagDataLength, newTag.data());
246  std::memcpy(originalTagData.Buffer + tagDataLength + newTag.size(), un.valueBuffer, sizeof(T));
247 
248  // store temp buffer back in TagData
249  const char* newTagData = (const char*)originalTagData.Buffer;
250  TagData.assign(newTagData, newTagDataLength);
251  return true;
252 }

312 template <typename T>
313 bool BamAlignment::AddTag(const std::string& tag, const std::vector<T>& values)
314 {
315 
316  // if char data not populated, do that first
317  if (SupportData.HasCoreOnly) BuildCharData();
318 
319  // check for valid tag name length
320  if (tag.size() != Constants::BAM_TAG_TAGSIZE) return false;
321 
322  // localize the tag data
323  char* pTagData = (char*)TagData.data();
324  const unsigned int tagDataLength = TagData.size();
325  unsigned int numBytesParsed = 0;
326 
327  // if tag already exists, return false
328  // use EditTag explicitly instead
329  if (FindTag(tag, pTagData, tagDataLength, numBytesParsed)) {
330  // TODO: set error string?
331  return false;
332  }
333 
334  // build new tag's base information
335  char newTagBase[Constants::BAM_TAG_ARRAYBASE_SIZE];
336  std::memcpy(newTagBase, tag.c_str(), Constants::BAM_TAG_TAGSIZE);
337  newTagBase[2] = Constants::BAM_TAG_TYPE_ARRAY;
338  newTagBase[3] = TagTypeHelper<T>::TypeCode();
339 
340  // add number of array elements to newTagBase
341  const int32_t numElements = values.size();
342  std::memcpy(newTagBase + 4, &numElements, sizeof(int32_t));
343 
344  // copy current TagData string to temp buffer, leaving room for new tag's contents
345  const std::size_t newTagDataLength =
346  tagDataLength + Constants::BAM_TAG_ARRAYBASE_SIZE + numElements * sizeof(T);
347  RaiiBuffer originalTagData(newTagDataLength);
348  std::memcpy(originalTagData.Buffer, TagData.c_str(),
349  tagDataLength + 1); // '+1' for TagData's null-term
350 
351  // write newTagBase (removes old null term)
352  std::strcat(originalTagData.Buffer + tagDataLength, (const char*)newTagBase);
353 
354  // add vector elements to tag
355  int elementsBeginOffset = tagDataLength + Constants::BAM_TAG_ARRAYBASE_SIZE;
356  for (int i = 0; i < numElements; ++i) {
357  const T& value = values.at(i);
358  std::memcpy(originalTagData.Buffer + elementsBeginOffset + i * sizeof(T), &value,
359  sizeof(T));
360  }
361 
362  // store temp buffer back in TagData
363  const char* newTagData = (const char*)originalTagData.Buffer;
364  TagData.assign(newTagData, newTagDataLength);
365  return true;
366 }

424 template <typename T>
425 bool BamAlignment::GetTag(const std::string& tag, T& destination) const
426 {
427 
428  // skip if alignment is core-only
429  if (SupportData.HasCoreOnly) {
430  // TODO: set error string?
431  return false;
432  }
433 
434  // skip if no tags present
435  if (TagData.empty()) {
436  // TODO: set error string?
437  return false;
438  }
439 
440  // localize the tag data
441  char* pTagData = (char*)TagData.data();
442  const unsigned int tagDataLength = TagData.size();
443  unsigned int numBytesParsed = 0;
444 
445  // return failure if tag not found
446  if (!FindTag(tag, pTagData, tagDataLength, numBytesParsed)) {
447  // TODO: set error string?
448  return false;
449  }
450 
451  // fetch data type
452  const char type = *(pTagData - 1);
453  if (!TagTypeHelper<T>::CanConvertFrom(type)) {
454  // TODO: set error string ?
455  return false;
456  }
457 
458  // determine data length
459  int destinationLength = 0;
460  switch (type) {
461 
462  // 1 byte data
463  case (Constants::BAM_TAG_TYPE_ASCII):
464  case (Constants::BAM_TAG_TYPE_INT8):
465  case (Constants::BAM_TAG_TYPE_UINT8):
466  destinationLength = 1;
467  break;
468 
469  // 2 byte data
470  case (Constants::BAM_TAG_TYPE_INT16):
471  case (Constants::BAM_TAG_TYPE_UINT16):
472  destinationLength = 2;
473  break;
474 
475  // 4 byte data
476  case (Constants::BAM_TAG_TYPE_INT32):
477  case (Constants::BAM_TAG_TYPE_UINT32):
478  case (Constants::BAM_TAG_TYPE_FLOAT):
479  destinationLength = 4;
480  break;
481 
482  // var-length types not supported for numeric destination
483  case (Constants::BAM_TAG_TYPE_STRING):
484  case (Constants::BAM_TAG_TYPE_HEX):
485  case (Constants::BAM_TAG_TYPE_ARRAY):
486  SetErrorString("BamAlignment::GetTag",
487  "cannot store variable length tag data into a numeric destination");
488  return false;
489 
490  // unrecognized tag type
491  default:
492  const std::string message = std::string("invalid tag type: ") + type;
493  SetErrorString("BamAlignment::GetTag", message);
494  return false;
495  }
496 
497  // store data in destination
498  destination = 0;
499  std::memcpy(&destination, pTagData, destinationLength);
500 
501  // return success
502  return true;
503 }

549 template <typename T>
550 bool BamAlignment::GetTag(const std::string& tag, std::vector<T>& destination) const
551 {
552 
553  // skip if alignment is core-only
554  if (SupportData.HasCoreOnly) {
555  // TODO: set error string?
556  return false;
557  }
558 
559  // skip if no tags present
560  if (TagData.empty()) {
561  // TODO: set error string?
562  return false;
563  }
564 
565  // localize the tag data
566  char* pTagData = (char*)TagData.data();
567  const unsigned int tagDataLength = TagData.size();
568  unsigned int numBytesParsed = 0;
569 
570  // return false if tag not found
571  if (!FindTag(tag, pTagData, tagDataLength, numBytesParsed)) {
572  // TODO: set error string?
573  return false;
574  }
575 
576  // check that tag is array type
577  const char tagType = *(pTagData - 1);
578  if (tagType != Constants::BAM_TAG_TYPE_ARRAY) {
579  SetErrorString("BamAlignment::GetTag", "cannot store a non-array tag in array destination");
580  return false;
581  }
582 
583  // fetch element type
584  const char elementType = *pTagData;
585  if (!TagTypeHelper<T>::CanConvertFrom(elementType)) {
586  // TODO: set error string ?
587  return false;
588  }
589  ++pTagData;
590 
591  // calculate length of each element in tag's array
592  switch (elementType) {
593  case (Constants::BAM_TAG_TYPE_ASCII):
594  case (Constants::BAM_TAG_TYPE_INT8):
595  case (Constants::BAM_TAG_TYPE_UINT8):
596  break;
597 
598  case (Constants::BAM_TAG_TYPE_INT16):
599  case (Constants::BAM_TAG_TYPE_UINT16):
600  break;
601 
602  case (Constants::BAM_TAG_TYPE_INT32):
603  case (Constants::BAM_TAG_TYPE_UINT32):
604  case (Constants::BAM_TAG_TYPE_FLOAT):
605  break;
606 
607  // var-length types not supported for numeric destination
608  case (Constants::BAM_TAG_TYPE_STRING):
609  case (Constants::BAM_TAG_TYPE_HEX):
610  case (Constants::BAM_TAG_TYPE_ARRAY):
611  SetErrorString("BamAlignment::GetTag",
612  "invalid array data, variable-length elements are not allowed");
613  return false;
614 
615  // unknown tag type
616  default:
617  const std::string message = std::string("invalid array element type: ") + elementType;
618  SetErrorString("BamAlignment::GetTag", message);
619  return false;
620  }
621 
622  // get number of elements
623  int32_t numElements;
624  std::memcpy(&numElements, pTagData, sizeof(int32_t));
625  pTagData += 4;
626  destination.clear();
627  destination.reserve(numElements);
628 
629  // read in elements
630  T value;
631  for (int i = 0; i < numElements; ++i) {
632  std::memcpy(&value, pTagData, sizeof(T));
633  pTagData += sizeof(T);
634  destination.push_back(value);
635  }
636 
637  // return success
638  return true;
639 }
