13 /* Illustrate the use of Z_BLOCK, inflatePrime(), and inflateSetDictionary()
14  for random access of a compressed file. A file containing a zlib or gzip
15  stream is provided on the command line. The compressed stream is decoded in
16  its entirety, and an index built with access points about every SPAN bytes
17  in the uncompressed output. The compressed file is left open, and can then
18  be read randomly, having to decompress on the average SPAN/2 uncompressed
19  bytes before getting to the desired block of data.
20 
21  An access point can be created at the start of any deflate block, by saving
22  the starting file offset and bit of that block, and the 32K bytes of
23  uncompressed data that precede that block. Also the uncompressed offset of
24  that block is saved to provide a referece for locating a desired starting
25  point in the uncompressed stream. build_index() works by decompressing the
26  input zlib or gzip stream a block at a time, and at the end of each block
27  deciding if enough uncompressed data has gone by to justify the creation of
28  a new access point. If so, that point is saved in a data structure that
29  grows as needed to accommodate the points.
30 
31  To use the index, an offset in the uncompressed data is provided, for which
32  the latest access point at or preceding that offset is located in the index.
33  The input file is positioned to the specified location in the index, and if
34  necessary the first few bits of the compressed data is read from the file.
35  inflate is initialized with those bits and the 32K of uncompressed data, and
36  the decompression then proceeds until the desired offset in the file is
37  reached. Then the decompression continues to read the desired uncompressed
38  data from the file.
39 
40  Another approach would be to generate the index on demand. In that case,
41  requests for random access reads from the compressed data would try to use
42  the index, but if a read far enough past the end of the index is required,
43  then further index entries would be generated and added.
44 
45  There is some fair bit of overhead to starting inflation for the random
46  access, mainly copying the 32K byte dictionary. So if small pieces of the
47  file are being accessed, it would make sense to implement a cache to hold
48  some lookahead and avoid many calls to extract() for small lengths.
49 
50  Another way to build an index would be to use inflateCopy(). That would
51  not be constrained to have access points at block boundaries, but requires
52  more memory per access point, and also cannot be saved to file due to the
53  use of pointers in the state. The approach here allows for storage of the
54  index in a file.
55  */

194 struct access *addpoint(struct access *index, int bits,
195  off_t in, off_t out, unsigned left, unsigned char *window)
196 {
197  struct point *next;
198 
199  /* if list is empty, create it (start with eight points) */
200  if (index == NULL) {
201  index = (struct access*) malloc(sizeof(struct access));
202  if (index == NULL) return NULL;
203  index->list = (struct point*) malloc(sizeof(struct point) << 3);
204  if (index->list == NULL) {
205  free(index);
206  return NULL;
207  }
208  index->size = 8;
209  index->have = 0;
210  }
211 
212  /* if list is full, make it bigger */
213  else if (index->have == index->size) {
214  index->size <<= 1;
215  next = (struct point*) realloc(index->list, sizeof(struct point) * index->size);
216  if (next == NULL) {
217  freeGzIndex(index);
218  return NULL;
219  }
220  index->list = next;
221  }
222 
223  /* fill in entry and increment how many we have */
224  next = index->list + index->have;
225  next->bits = bits;
226  next->in = in;
227  next->out = out;
228  if (left)
229  memcpy(next->window, window + WINSIZE - left, left);
230  if (left < WINSIZE)
231  memcpy(next->window + left, window, WINSIZE - left);
232  index->have++;
233 
234  /* return list, possibly reallocated */
235  return index;
236 }

238 /* Make one entire pass through the compressed stream and build an index, with
239  access points about every span bytes of uncompressed output -- span is
240  chosen to balance the speed of random access against the memory requirements
241  of the list, about 32K bytes per access point. Note that data after the end
242  of the first zlib or gzip stream in the file is ignored. build_index()
243  returns the number of access points on success (>= 1), Z_MEM_ERROR for out
244  of memory, Z_DATA_ERROR for an error in the input file, or Z_ERRNO for a
245  file read error. On success, *built points to the resulting index. */
246 int buildGzIndex_Stream(FILE *in, off_t span, struct access **built)
247 {
248  int ret;
249  off_t totin, totout; /* our own total counters to avoid 4GB limit */
250  off_t last; /* totout value of last access point */
251  struct access *index; /* access points being generated */
252  z_stream strm;
253  unsigned char input[CHUNK];
254  unsigned char window[WINSIZE];
255 
256  /* initialize inflate */
257  strm.zalloc = Z_NULL;
258  strm.zfree = Z_NULL;
259  strm.opaque = Z_NULL;
260  strm.avail_in = 0;
261  strm.next_in = Z_NULL;
262  ret = inflateInit2(&strm, 47); /* automatic zlib or gzip decoding */
263  if (ret != Z_OK)
264  return ret;
265 
266  /* inflate the input, maintain a sliding window, and build an index -- this
267  also validates the integrity of the compressed data using the check
268  information at the end of the gzip or zlib stream */
269  totin = totout = last = 0;
270  index = NULL; /* will be allocated by first addpoint() */
271  strm.avail_out = 0;
272  do {
273  /* get some compressed data from input file */
274  strm.avail_in = fread(input, 1, CHUNK, in);
275  if (ferror(in)) {
276  ret = Z_ERRNO;
277  goto build_index_error;
278  }
279  if (strm.avail_in == 0) {
280  ret = Z_DATA_ERROR;
281  goto build_index_error;
282  }
283  strm.next_in = input;
284 
285  /* process all of that, or until end of stream */
286  do {
287  /* reset sliding window if necessary */
288  if (strm.avail_out == 0) {
289  strm.avail_out = WINSIZE;
290  strm.next_out = window;
291  }
292 
293  /* inflate until out of input, output, or at end of block --
294  update the total input and output counters */
295  totin += strm.avail_in;
296  totout += strm.avail_out;
297  ret = inflate(&strm, Z_BLOCK); /* return at end of block */
298  totin -= strm.avail_in;
299  totout -= strm.avail_out;
300  if (ret == Z_NEED_DICT)
301  ret = Z_DATA_ERROR;
302  if (ret == Z_MEM_ERROR || ret == Z_DATA_ERROR)
303  goto build_index_error;
304  if (ret == Z_STREAM_END)
305  break;
306 
307  /* if at end of block, consider adding an index entry (note that if
308  data_type indicates an end-of-block, then all of the
309  uncompressed data from that block has been delivered, and none
310  of the compressed data after that block has been consumed,
311  except for up to seven bits) -- the totout == 0 provides an
312  entry point after the zlib or gzip header, and assures that the
313  index always has at least one access point; we avoid creating an
314  access point after the last block by checking bit 6 of data_type
315  */
316  if ((strm.data_type & 128) && !(strm.data_type & 64) &&
317  (totout == 0 || totout - last > span)) {
318  index = addpoint(index, strm.data_type & 7, totin,
319  totout, strm.avail_out, window);
320  if (index == NULL) {
321  ret = Z_MEM_ERROR;
322  goto build_index_error;
323  }
324  last = totout;
325  }
326  } while (strm.avail_in != 0);
327  } while (ret != Z_STREAM_END);
328 
329  /* clean up and return index (release unused entries in list) */
330  (void)inflateEnd(&strm);
331  index->list = (struct point*) realloc(index->list, sizeof(struct point) * index->have);
332  index->size = index->have;
333  *built = index;
334  return index->size;
335 
336  /* return error */
337  build_index_error:
338  (void)inflateEnd(&strm);
339  if (index != NULL)
340  freeGzIndex(index);
341  return ret;
342 }

358 /* Use the index to read len bytes from offset into buf, return bytes read or
359  negative for error (Z_DATA_ERROR or Z_MEM_ERROR). If data is requested past
360  the end of the uncompressed data, then extract() will return a value less
361  than len, indicating how much as actually read into buf. This function
362  should not return a data error unless the file was modified since the index
363  was generated. extract() may also return Z_ERRNO if there is an error on
364  reading or seeking the input file. */
365 int extract_Stream(FILE* in, struct access *index, off_t offset,
366  unsigned char *buf, int len)
367 {
368  int ret, skip;
369  z_stream strm;
370  struct point *here;
371  unsigned char input[CHUNK];
372  unsigned char discard[WINSIZE];
373 
374  /* proceed only if something reasonable to do */
375  if (len < 0)
376  return 0;
377 
378  /* find where in stream to start */
379  here = index->list;
380  ret = index->have;
381  while (--ret && here[1].out <= offset)
382  here++;
383 
384  /* initialize file and inflate state to start there */
385  strm.zalloc = Z_NULL;
386  strm.zfree = Z_NULL;
387  strm.opaque = Z_NULL;
388  strm.avail_in = 0;
389  strm.next_in = Z_NULL;
390  ret = inflateInit2(&strm, -15); /* raw inflate */
391  if (ret != Z_OK)
392  return ret;
393  ret = fseeko(in, here->in - (here->bits ? 1 : 0), SEEK_SET);
394  if (ret == -1)
395  goto extract_ret;
396  if (here->bits) {
397  ret = getc(in);
398  if (ret == -1) {
399  ret = ferror(in) ? Z_ERRNO : Z_DATA_ERROR;
400  goto extract_ret;
401  }
402  (void)inflatePrime(&strm, here->bits, ret >> (8 - here->bits));
403  }
404  (void)inflateSetDictionary(&strm, here->window, WINSIZE);
405 
406  /* skip uncompressed bytes until offset reached, then satisfy request */
407  offset -= here->out;
408  strm.avail_in = 0;
409  skip = 1; /* while skipping to offset */
410  do {
411  /* define where to put uncompressed data, and how much */
412  if (offset == 0 && skip) { /* at offset now */
413  strm.avail_out = len;
414  strm.next_out = buf;
415  skip = 0; /* only do this once */
416  }
417  if (offset > WINSIZE) { /* skip WINSIZE bytes */
418  strm.avail_out = WINSIZE;
419  strm.next_out = discard;
420  offset -= WINSIZE;
421  }
422  else if (offset != 0) { /* last skip */
423  strm.avail_out = (unsigned)offset;
424  strm.next_out = discard;
425  offset = 0;
426  }
427 
428  /* uncompress until avail_out filled, or end of stream */
429  do {
430  if (strm.avail_in == 0) {
431  strm.avail_in = fread(input, 1, CHUNK, in);
432  if (ferror(in)) {
433  ret = Z_ERRNO;
434  goto extract_ret;
435  }
436  if (strm.avail_in == 0) {
437  ret = Z_DATA_ERROR;
438  goto extract_ret;
439  }
440  strm.next_in = input;
441  }
442  ret = inflate(&strm, Z_NO_FLUSH); /* normal inflate */
443  if (ret == Z_NEED_DICT)
444  ret = Z_DATA_ERROR;
445  if (ret == Z_MEM_ERROR || ret == Z_DATA_ERROR)
446  goto extract_ret;
447  if (ret == Z_STREAM_END)
448  break;
449  } while (strm.avail_out != 0);
450 
451  /* if reach end of stream, then don't keep trying to get more */
452  if (ret == Z_STREAM_END)
453  break;
454 
455  /* do until offset reached and requested data read, or stream ends */
456  } while (skip);
457 
458  /* compute number of uncompressed bytes read after offset */
459  ret = skip ? 0 : len - strm.avail_out;
460 
461  /* clean up and return bytes read or error */
462  extract_ret:
463  (void)inflateEnd(&strm);
464  return ret;
465 }
